.remove([selector])


Description

With selector: Removes any matched elements from the DOM that are children of any of the elements in the collection

Without selector: Removes each element from the DOM and the collection


.empty()


Description

Removes the innerHTML of each element in the collection.


Uses

HTML

<h1 id="will-empty">
  This text will be removed but the h1 tag will remain.
</h1>

CSS

#will-empty {
  font-weight: 700;
}

jaysQuery

$j('#will-empty').empty();

This text will be removed but the h1 tag will remain.

.off(event)


Description

Detatches event hanndler functions for each element in the collection.


Uses

HTML

<figure class="lock-button">LOCK</figure>
<figure id="moveable-box" class="first-pos"></figure>

CSS

#moveable-box {
  background-color: red;
  width: 50px;
  height: 50px;
  position: relative;
  transition: left 0.5s;
}

.first-pos {
  left: 0px;
}

.second-pos {
  left: 50px;
}

.third-pos {
  left: 100px;
}

.fourth-pos {
  left: 150px;
}

.fifth-pos {
  left: 200px;
}

.sixth-pos {
  left: 250px;
}

.lock-button {
  border: none;
  padding: 4px 6px;
  font-size: 15px;
  font-weight: 700;
  border-radius: 6px;
  color: #41baf8;
  background-color: #4170f8;
  align-self: center;

jaysQuery

const checkBox = (e) => {
  switch (e.currentTarget.innerHTML) {
    case "LOCK":
      $moveableBox.css('background-color', 'blue');
      $moveableBox.off('mouseenter');
      $lockButton.html('UNLOCK');
      return;
    case "UNLOCK":
      $moveableBox.css('background-color', 'red');
      $moveableBox.on('mouseenter', changeButtonPosition);
      $lockButton.html('LOCK');
      return;
    default:
      return;
  }
};

const changeButtonPosition = (e) => {
  switch (e.target.className) {
    case FIRST:
      $moveableBox.removeClass('first-pos');
      $moveableBox.addClass('second-pos');
      return;
  moveableBox  case SECOND:
      $moveableBox.removeClass('second-pos');
      $moveableBox.addClass('third-pos');
      return;
    case THIRD:
      $moveableBox.removeClass('third-pos');
      $moveableBox.addClass('fourth-pos');
      return;
    case FOURTH:
      $moveableBox.removeClass('fourth-pos');
      $moveableBox.addClass('fifth-pos');
      return;
    case FIFTH:
      $moveableBox.removeClass('fifth-pos');
      $moveableBox.addClass('sixth-pos');
      return;
    case SIXTH:
      $moveableBox.removeClass('sixth-pos');
      $moveableBox.addClass('first-pos');

      return;
  }
};

$lockButton = $j('.lock-button');
$lockButton.on('click', checkBox);
$moveableBox = $j('#moveable-box');
$moveableBox.on('mouseenter', changeButtonPosition);

.on(event, callback)


Description

Executes callback whenever event occurs to any element in the set of elements by attaching event handler functions.


Uses

  font-size: 24.5px;

  margin-bottom: -30px;;

.html([string])


Description

With Argument: Replaces the innerHTML of each element in the collection

Without Argument: Returns the innerHTML of the first element in the collection


Uses

HTML

<label id="html-label" for="html-input">
  Enter Text:
</label>
<br>
<input id="html-input" type="text" value="Example Text">
<br>
<p class="change-html"></p>

CSS

#html-label {
  font-weight: 700;
}
#html-input {
  font-size: 20px;
}
.change-html {
  min-width: 227px;
  min-height: 25px;
  border: 1px solid black;
  font-size: 20px;
  display: inline-block;
}

jaysQuery

let input, $paragraph;
$j('#html-input').each((node) => input = node.value);
$paragraph = $j('.change-html');
$paragraph.html(input);

.show()


Description

Shows a hidden element

Styles are restored to whatever they were initially.


Uses

HTML

<p id="show-text" class="hidden">
  POOF!
</p>

CSS

.on-show {
  color: green;
}

.hidden {
  display: none;
}

jaysQuery

$j('#show-text').show();

POOF!

.hide()


Description

Hides the elements in the collection

To make this work you must include a class named .hidden { display: none; } in your stylesheet


Uses

HTML

<p class="will-disappear">
  Test the magic.
</p>

CSS

.will-disappear {
  font-weight: 700;
}

.hidden {
  display: none;
}

jaysQuery

$j('.will-disappear').hide();

Test the magic.

.each(callback)


Description

For each element/node, executes the callback.


Uses

HTML

<p class="to-li">
  Text will turn into a blue LI.
</p>

<div class="to-li">
  Text will also turn into a blue LI.
</div>

CSS

.to-li {
  font-weight: 700;
}

.new-li {
  color: blue;
}

jaysQuery

$j('.to-li').each((el) => el.outerHTML = (`<li class="new-li">${el.innerHTML}</li>`);

Text will turn into a blue LI.

Text will also turn into a blue LI.

.removeClass(className)


Description

Removes the specified class from each element in the set of matched elements.


Uses

HTML

<p id="sentence" class="red">
  Text will turn black.
</p>

CSS

#sentence {
  font-weight: 700;
}

.red {
  color: red;
}

jaysQuery

$j('#sentence').removeClass('red');

Text will turn black.

.addClass(className)


Description

Adds the specified class(es) to each element in the set of matched elements.

This method does not replace a class. It appends it to the given classes of the node collection.


Uses

HTML

<p id="text" class="">
  Text will turn red
</p>

CSS

#text {
  font-weight: 700;
}

.red {
  color: red;
}

jaysQuery

$j('#text').addClass('red');

Text will turn red

What is jaysQuery?

jaysQuery is a small yet versatile JavaScript library. It was created by Justin Cox in 2018 to assist with DOM traversal and manipulation, event-handling, animation, and making AJAX requests.
